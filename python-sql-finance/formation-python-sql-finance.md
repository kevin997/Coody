```python
montant_final = capital + interets
```

### 2.6 Exemple financier : Calculateur d'impôts progressifs

```python
"""
Barème d'imposition :
- 0 à 500,000 : 0%
- 500,001 à 1,000,000 : 10%
- 1,000,001 à 2,000,000 : 20%
- Au-delà de 2,000,000 : 30%
"""

revenu = float(input("Entrez votre revenu annuel (FCFA) : "))

if revenu <= 500000:
    taux = 0
    impot = 0
elif revenu <= 1000000:
    taux = 0.10
    impot = (revenu - 500000) * taux
elif revenu <= 2000000:
    taux = 0.20
    # 10% sur la tranche 500k-1M + 20% sur le reste
    impot = (500000 * 0.10) + ((revenu - 1000000) * taux)
else:
    taux = 0.30
    # Calcul sur toutes les tranches
    impot = (500000 * 0.10) + (1000000 * 0.20) + ((revenu - 2000000) * taux)

revenu_net = revenu - impot
taux_effectif = (impot / revenu) * 100

print("\n" + "="*50)
print("CALCUL D'IMPÔT")
print("="*50)
print(f"Revenu brut      : {revenu:,.0f} FCFA")
print(f"Impôt dû         : {impot:,.0f} FCFA")
print(f"Revenu net       : {revenu_net:,.0f} FCFA")
print(f"Taux effectif    : {taux_effectif:.2f}%")
print("="*50)
```

## Chapitre 3 : Boucles

### 3.1 La boucle while

```python
# Boucle tant que la condition est vraie
compteur = 1

while compteur <= 5:
    print(f"Itération {compteur}")
    compteur += 1  # Équivalent à : compteur = compteur + 1

print("Boucle terminée")
```

**Attention :** Assurez-vous que la condition devient fausse, sinon boucle infinie !

### 3.2 La boucle for

```python
# Boucle pour un nombre fixe d'itérations
for i in range(5):
    print(f"Itération {i}")

# range(start, stop, step)
for i in range(1, 10, 2):  # 1, 3, 5, 7, 9
    print(i)

# Parcourir une liste
actions = ["AAPL", "GOOGL", "MSFT"]
for action in actions:
    print(f"Ticker : {action}")
```

### 3.3 Exemple financier : Intérêts composés

```python
"""
Formule : Montant_final = Principal × (1 + taux)^temps
"""

capital = 10000
```

### 3.4 break et continue

```python
# break : sortir de la boucle
investissement = 10000
objectif = 50000
annee = 0
taux = 0.10

while True:  # Boucle infinie
    annee += 1
    investissement *= (1 + taux)
    
    if investissement >= objectif:
        print(f"Objectif atteint en {annee} ans !")
        break  # Sortir de la boucle

# continue : passer à l'itération suivante
for i in range(1, 11):
    if i % 2 == 0:  # Si pair
        continue  # Ignorer les pairs
    print(i)  # Affiche seulement les impairs
```

### 3.5 Exemple : Tableau d'amortissement

```python
"""
Créer un tableau d'amortissement pour un prêt
"""

duree_annees = 5

# Calcul de la mensualité
taux_mensuel = taux_annuel / 12
nombre_paiements = duree_annees * 12

mensualite = (montant_pret * taux_mensuel * (1 + taux_mensuel)**nombre_paiements) / \
             ((1 + taux_mensuel)**nombre_paiements - 1)

solde = montant_pret

print(f"Montant du prêt : {montant_pret:,.0f} FCFA")
```
